\documentclass[tikz,border=10pt]{standalone}
\usepackage{tikz}
\usetikzlibrary{arrows.meta}

\begin{document}
\begin{tikzpicture}[
  >=Latex,
  thick,
  box/.style={draw, rounded corners=5pt, minimum width=3.2cm, minimum height=1.1cm, font=\large}
]

\node[box] (agent) at (0, 3)  {Agent};
\node[box] (env)   at (0, 0)  {Environment};

% Right path (action)
\draw[->] (agent.east) -- ++(2.8,0) -- ++(0,-3) -- (env.east);
\node[right, align=left, font=\normalsize] at (2.8, 1.5) {action\\[2pt]$A_t$};

% Left path (state + reward)
\draw[->] (env.west) -- ++(-2.8,0) -- ++(0,3) -- (agent.west);
\node[left,  align=right, font=\normalsize] at (-2.8, 3)   {state\\[2pt]$S_t$};
\node[right, align=left,  font=\normalsize] at (-2.8, 1.5) {reward\\[2pt]$R_t$};

% Dashed line with next-step labels
\draw[dashed, thin] (-1.0, 0.55) -- (-1.0, -0.55);
\node[right, font=\small] at (-1.0, 0.45) {$R_{t+1}$};
\node[right, font=\small] at (-1.0, 0.15) {$S_{t+1}$};

\end{tikzpicture}
